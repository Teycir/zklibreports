    2: pragma solidity 0.8.24;
    3: 
    4: import { BitMaps } from "@openzeppelin/contracts/utils/structs/BitMaps.sol";
    5: import { L1MessageManagerV1 } from "./v1/L1MessageManagerV1.sol";
    6: import { IL1MessageManager } from "../../interfaces/l1/IL1MessageManager.sol";
    7: import { Utils } from "../../lib/Utils.sol";
    8: 
    9: /**
   10:  * @title Contract to manage cross-chain message rolling hash computation and storage on L1.
   11:  * @author ConsenSys Software Inc.
   12:  * @custom:security-contact security-report@linea.build
   13:  */
   14: abstract contract L1MessageManager is L1MessageManagerV1, IL1MessageManager {
   15:   using BitMaps for BitMaps.BitMap;
   16:   using Utils for *;
   17: 
   18:   mapping(uint256 messageNumber => bytes32 rollingHash) public rollingHashes;
   19:   BitMaps.BitMap internal _messageClaimedBitMap;
   20:   mapping(bytes32 merkleRoot => uint256 treeDepth) public l2MerkleRootsDepths;
   21: 
   22:   /// @dev Total contract storage is 53 slots including the gap below.
   23:   /// @dev Keep 50 free storage slots for future implementation updates to avoid storage collision.
   24:   uint256[50] private __gap_L1MessageManager;
   25: 
   26:   /**
   27:    * @notice Take an existing message hash, calculates the rolling hash and stores at the message number.
   28:    * @param _messageNumber The current message number being sent.
   29:    * @param _messageHash The hash of the message being sent.
   30:    */
   31:   function _addRollingHash(uint256 _messageNumber, bytes32 _messageHash) internal {
   32:     unchecked {
   33:       bytes32 newRollingHash = Utils._efficientKeccak(rollingHashes[_messageNumber - 1], _messageHash);
   34: 
   35:       rollingHashes[_messageNumber] = newRollingHash;
   36:       emit RollingHashUpdated(_messageNumber, newRollingHash, _messageHash);
   37:     }
   38:   }
   39: 
   40:   /**
   41:    * @notice Set the L2->L1 message as claimed when a user claims a message on L1.
   42:    * @param  _messageNumber The message number on L2.
   43:    */
   44:   function _setL2L1MessageToClaimed(uint256 _messageNumber) internal {
   45:     if (_messageClaimedBitMap.get(_messageNumber)) {
   46:       revert MessageAlreadyClaimed(_messageNumber);
   47:     }
   48:     _messageClaimedBitMap.set(_messageNumber);
   49:   }
   50: 
   51:   /**
   52:    * @notice Add the L2 merkle roots to the storage.
   53:    * @dev This function is called during block finalization.
   54:    * @dev The _treeDepth does not need to be checked to be non-zero as it is,
   55:    * already enforced to be non-zero in the circuit, and used in the proof's public input.
   56:    * @param _newRoots New L2 merkle roots.
   57:    */
   58:   function _addL2MerkleRoots(bytes32[] calldata _newRoots, uint256 _treeDepth) internal {
   59:     for (uint256 i; i < _newRoots.length; ++i) {
   60:       if (l2MerkleRootsDepths[_newRoots[i]] != 0) {
   61:         revert L2MerkleRootAlreadyAnchored(_newRoots[i]);
   62:       }
   63: 
   64:       l2MerkleRootsDepths[_newRoots[i]] = _treeDepth;
   65: 
   66:       emit L2MerkleRootAdded(_newRoots[i], _treeDepth);
   67:     }
   68:   }
   69: 
   70:   /**
   71:    * @notice Emit an event for each L2 block containing L2->L1 messages.
   72:    * @dev This function is called during block finalization.
   73:    * @param _l2MessagingBlocksOffsets Is a sequence of uint16 values, where each value plus the last finalized L2 block number.
   74:    * indicates which L2 blocks have L2->L1 messages.
   75:    * @param _currentL2BlockNumber Last L2 block number finalized on L1.
   76:    */
   77:   function _anchorL2MessagingBlocks(bytes calldata _l2MessagingBlocksOffsets, uint256 _currentL2BlockNumber) internal {
   78:     if (_l2MessagingBlocksOffsets.length % 2 != 0) {
   79:       revert BytesLengthNotMultipleOfTwo(_l2MessagingBlocksOffsets.length);
   80:     }
   81: 
   82:     uint256 l2BlockOffset;
   83:     unchecked {
   84:       for (uint256 i; i < _l2MessagingBlocksOffsets.length; ) {
   85:         assembly {
   86:           l2BlockOffset := shr(240, calldataload(add(_l2MessagingBlocksOffsets.offset, i)))
   87:         }
   88:         emit L2MessagingBlockAnchored(_currentL2BlockNumber + l2BlockOffset);
   89:         i += 2;
   90:       }
   91:     }
   92:   }
   93: 
   94:   /**
   95:    * @notice Check if the L2->L1 message is claimed or not.
   96:    * @param _messageNumber The message number on L2.
   97:    */
   98:   function isMessageClaimed(uint256 _messageNumber) external view returns (bool) {
   99:     return _messageClaimedBitMap.get(_messageNumber);
  100:   }
  101: }
