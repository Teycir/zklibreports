   36: /**
   37:  * @dev Contract module which provides a basic access control mechanism, where
   38:  * there is an account (an owner) that can be granted exclusive access to
   39:  * specific functions.
   40:  *
   41:  * By default, the owner account will be the one that deploys the contract. This
   42:  * can later be changed with {transferOwnership}.
   43:  *
   44:  * This module is used through inheritance. It will make available the modifier
   45:  * `onlyOwner`, which can be applied to your functions to restrict their use to
   46:  * the owner.
   47:  */
   48: abstract contract Ownable is Context {
   49:   address private _owner;
   50: 
   51:   event OwnershipTransferred(address indexed previousOwner, address indexed newOwner);
   52: 
   53:   /**
   54:    * @dev Initializes the contract setting the deployer as the initial owner.
   55:    */
   56:   constructor() {
   57:     _transferOwnership(_msgSender());
   58:   }
   59: 
   60:   /**
   61:    * @dev Throws if called by any account other than the owner.
   62:    */
   63:   modifier onlyOwner() {
   64:     _checkOwner();
   65:     _;
   66:   }
   67: 
   68:   /**
   69:    * @dev Returns the address of the current owner.
   70:    */
   71:   function owner() public view virtual returns (address) {
   72:     return _owner;
   73:   }
   74: 
   75:   /**
   76:    * @dev Throws if the sender is not the owner.
   77:    */
   78:   function _checkOwner() internal view virtual {
   79:     require(owner() == _msgSender(), "Ownable: caller is not the owner");
   80:   }
   81: 
   82:   /**
   83:    * @dev Leaves the contract without owner. It will not be possible to call
   84:    * `onlyOwner` functions. Can only be called by the current owner.
   85:    *
   86:    * NOTE: Renouncing ownership will leave the contract without an owner,
   87:    * thereby disabling any functionality that is only available to the owner.
   88:    */
   89:   function renounceOwnership() public virtual onlyOwner {
   90:     _transferOwnership(address(0));
   91:   }
   92: 
   93:   /**
   94:    * @dev Transfers ownership of the contract to a new account (`newOwner`).
   95:    * Can only be called by the current owner.
   96:    */
   97:   function transferOwnership(address newOwner) public virtual onlyOwner {
   98:     require(newOwner != address(0), "Ownable: new owner is the zero address");
   99:     _transferOwnership(newOwner);
  100:   }
  101: 
  102:   /**
  103:    * @dev Transfers ownership of the contract to a new account (`newOwner`).
  104:    * Internal function without access restriction.
  105:    */
  106:   function _transferOwnership(address newOwner) internal virtual {
  107:     address oldOwner = _owner;
  108:     _owner = newOwner;
  109:     emit OwnershipTransferred(oldOwner, newOwner);
  110:   }
  111: }

  195:    * ====

  197:    * false is an externally-owned account (EOA) and not a contract.
  198:    *
  199:    * Among others, `isContract` will return false for the following
  200:    * types of addresses:
  201:    *
  202:    *  - an externally-owned account
  203:    *  - a contract in construction
  204:    *  - an address where a contract will be created
  205:    *  - an address where a contract lived, but was destroyed
  206:    *
  207:    * Furthermore, `isContract` will also return true if the target contract within
  208:    * the same transaction is already scheduled for destruction by `SELFDESTRUCT`,
  209:    * which only has an effect at the end of a transaction.
  210:    * ====
  211:    *
  212:    * [IMPORTANT]
  213:    * ====
  214:    * You shouldn't rely on `isContract` to protect against flash loan attacks!
  215:    *
  216:    * Preventing calls from contracts is highly discouraged. It breaks composability, breaks support for smart wallets
  217:    * like Gnosis Safe, and does not provide security since it can be circumvented by calling from a contract
  218:    * constructor.

  220:    */

  366:   /**
  367:    * @dev Tool to verify that a low level call to smart-contract was successful, and revert (either by bubbling
  368:    * the revert reason or using the provided one) in case of unsuccessful call or if target was not a contract.
  369:    *
  370:    * _Available since v4.8._

  380:         // only check isContract if the call was successful and the return data is empty
  381:         // otherwise we already know that it was a contract
  382:         require(isContract(target), "Address: call to non-contract");
  383:       }
  384:       return returndata;
  385:     } else {
  386:       _revert(returndata, errorMessage);
  387:     }
  388:   }
  389: 
  390:   /**
  391:    * @dev Tool to verify that a low level call was successful, and revert if it wasn't, either by bubbling the
  392:    * revert reason or using the provided one.
  393:    *
  394:    * _Available since v4.3._
  395:    */
  396:   function verifyCallResult(
  397:     bool success,
  398:     bytes memory returndata,
  399:     string memory errorMessage
  400:   ) internal pure returns (bytes memory) {
  401:     if (success) {
  402:       return returndata;
  403:     } else {
  404:       _revert(returndata, errorMessage);
  405:     }
  406:   }
  407: 
  408:   function _revert(bytes memory returndata, string memory errorMessage) private pure {
  409:     // Look for revert reason and bubble it up if present
  410:     if (returndata.length > 0) {
  411:       // The easiest way to bubble the revert reason is using memory via assembly
  412:       /// @solidity memory-safe-assembly
  413:       assembly {
  414:         let returndata_size := mload(returndata)
  415:         revert(add(32, returndata), returndata_size)
  416:       }
  417:     } else {
  418:       revert(errorMessage);
  419:     }
  420:   }
  421: }

  722: /**
  723:  * @dev This abstract contract provides a fallback function that delegates all calls to another contract using the EVM
  724:  * instruction `delegatecall`. We refer to the second contract as the _implementation_ behind the proxy, and it has to
  725:  * be specified by overriding the virtual {_implementation} function.
  726:  *
  727:  * Additionally, delegation to the implementation can be triggered manually through the {_fallback} function, or to a
  728:  * different contract through the {_delegate} function.
  729:  *
  730:  * The success and return data of the delegated call will be returned back to the caller of the proxy.
  731:  */
  732: abstract contract Proxy {
  733:   /**
  734:    * @dev Delegates the current call to `implementation`.
  735:    *
  736:    * This function does not return to its internal call site, it will return directly to the external caller.
  737:    */
  738:   function _delegate(address implementation) internal virtual {
  739:     assembly {
  740:       // Copy msg.data. We take full control of memory in this inline assembly
  741:       // block because it will not return to Solidity code. We overwrite the
  742:       // Solidity scratch pad at memory position 0.
  743:       calldatacopy(0, 0, calldatasize())
  744: 
  745:       // Call the implementation.
  746:       // out and outsize are 0 because we don't know the size yet.
  747:       let result := delegatecall(gas(), implementation, 0, calldatasize(), 0, 0)
  748: 
  749:       // Copy the returned data.
  750:       returndatacopy(0, 0, returndatasize())
  751: 
  752:       switch result
  753:       // delegatecall returns 0 on error.
  754:       case 0 {
  755:         revert(0, returndatasize())
  756:       }
  757:       default {
  758:         return(0, returndatasize())
  759:       }
  760:     }
  761:   }

  768: 
  769:   /**
  770:    * @dev Delegates the current call to the address returned by `_implementation()`.
  771:    *
  772:    * This function does not return to its internal call site, it will return directly to the external caller.
  773:    */

  775:     _beforeFallback();
  776:     _delegate(_implementation());
  777:   }

  865:  * things that go hand in hand:
  866:  *
  867:  * 1. If any account other than the admin calls the proxy, the call will be forwarded to the implementation, even if
  868:  * that call matches one of the admin functions exposed by the proxy itself.
  869:  * 2. If the admin calls the proxy, it can access the admin functions, but its calls will never be forwarded to the
  870:  * implementation. If the admin tries to call a function on the implementation it will fail with an error that says
  871:  * "admin cannot fallback to proxy target".
  872:  *
  873:  * These properties mean that the admin account can only be used for admin actions like upgrading the proxy or changing
  874:  * the admin, so it's best if it's a dedicated account that is not used for anything else. This will avoid headaches due
  875:  * to sudden errors when trying to call a function from the proxy implementation.

  878:  * you should think of the `ProxyAdmin` instance as the real administrative interface of your proxy.
