   89:    *
   90:    * Among others, `isContract` will return false for the following
   91:    * types of addresses:
   92:    *

  101:    * ====
  102:    *
  103:    * [IMPORTANT]

  105:    * You shouldn't rely on `isContract` to protect against flash loan attacks!
  106:    *
  107:    * Preventing calls from contracts is highly discouraged. It breaks composability, breaks support for smart wallets

  273:         require(isContract(target), "Address: call to non-contract");
  274:       }
  275:       return returndata;
  276:     } else {
  277:       _revert(returndata, errorMessage);
  278:     }
  279:   }
  280: 
  281:   /**
  282:    * @dev Tool to verify that a low level call was successful, and revert if it wasn't, either by bubbling the
  283:    * revert reason or using the provided one.
  284:    *
  285:    * _Available since v4.3._
  286:    */
  287:   function verifyCallResult(
  288:     bool success,
  289:     bytes memory returndata,
  290:     string memory errorMessage
  291:   ) internal pure returns (bytes memory) {
  292:     if (success) {
  293:       return returndata;
  294:     } else {
  295:       _revert(returndata, errorMessage);
  296:     }
  297:   }
  298: 
  299:   function _revert(bytes memory returndata, string memory errorMessage) private pure {
  300:     // Look for revert reason and bubble it up if present
  301:     if (returndata.length > 0) {
  302:       // The easiest way to bubble the revert reason is using memory via assembly
  303:       /// @solidity memory-safe-assembly
  304:       assembly {
  305:         let returndata_size := mload(returndata)
  306:         revert(add(32, returndata), returndata_size)
  307:       }
  308:     } else {
  309:       revert(errorMessage);
  310:     }
  311:   }
  312: }

  613: /**
  614:  * @dev This abstract contract provides a fallback function that delegates all calls to another contract using the EVM
  615:  * instruction `delegatecall`. We refer to the second contract as the _implementation_ behind the proxy, and it has to
  616:  * be specified by overriding the virtual {_implementation} function.
  617:  *
  618:  * Additionally, delegation to the implementation can be triggered manually through the {_fallback} function, or to a
  619:  * different contract through the {_delegate} function.
  620:  *
  621:  * The success and return data of the delegated call will be returned back to the caller of the proxy.
  622:  */
  623: abstract contract Proxy {
  624:   /**
  625:    * @dev Delegates the current call to `implementation`.
  626:    *
  627:    * This function does not return to its internal call site, it will return directly to the external caller.
  628:    */
  629:   function _delegate(address implementation) internal virtual {
  630:     assembly {
  631:       // Copy msg.data. We take full control of memory in this inline assembly
  632:       // block because it will not return to Solidity code. We overwrite the
  633:       // Solidity scratch pad at memory position 0.
  634:       calldatacopy(0, 0, calldatasize())
  635: 
  636:       // Call the implementation.
  637:       // out and outsize are 0 because we don't know the size yet.
  638:       let result := delegatecall(gas(), implementation, 0, calldatasize(), 0, 0)
  639: 
  640:       // Copy the returned data.
  641:       returndatacopy(0, 0, returndatasize())
  642: 
  643:       switch result
  644:       // delegatecall returns 0 on error.
  645:       case 0 {
  646:         revert(0, returndatasize())
  647:       }
  648:       default {
  649:         return(0, returndatasize())
  650:       }
  651:     }
  652:   }
  653: 
  654:   /**
  655:    * @dev This is a virtual function that should be overridden so it returns the address to which the fallback function
  656:    * and {_fallback} should delegate.
  657:    */
  658:   function _implementation() internal view virtual returns (address);
  659: 
  660:   /**
  661:    * @dev Delegates the current call to the address returned by `_implementation()`.
  662:    *
  663:    * This function does not return to its internal call site, it will return directly to the external caller.
  664:    */
  665:   function _fallback() internal virtual {
  666:     _beforeFallback();
  667:     _delegate(_implementation());
  668:   }
  669: 
  670:   /**
  671:    * @dev Fallback function that delegates calls to the address returned by `_implementation()`. Will run if no other
  672:    * function in the contract matches the call data.
  673:    */
  674:   fallback() external payable virtual {
  675:     _fallback();
  676:   }
  677: 
  678:   /**
  679:    * @dev Fallback function that delegates calls to the address returned by `_implementation()`. Will run if call data
  680:    * is empty.
  681:    */
  682:   receive() external payable virtual {
  683:     _fallback();
  684:   }
  685: 
  686:   /**

  688:    * call, or as part of the Solidity `fallback` or `receive` functions.
  689:    *
  690:    * If overridden should call `super._beforeFallback()`.
  691:    */
  692:   function _beforeFallback() internal virtual {}

  735:  * mechanism. The compiler is unaware that these functions are implemented by {TransparentUpgradeableProxy} and will not

  738: interface ITransparentUpgradeableProxy is IERC1967 {
  739:   function admin() external view returns (address);

  750: /**
  751:  * @dev This contract implements a proxy that is upgradeable by an admin.
  752:  *
  753:  * To avoid https://medium.com/nomic-labs-blog/malicious-backdoors-in-ethereum-proxies-62629adf3357[proxy selector
  754:  * clashing], which can potentially be used in an attack, this contract uses the
  755:  * https://blog.openzeppelin.com/the-transparent-proxy-pattern/[transparent proxy pattern]. This pattern implies two
  756:  * things that go hand in hand:
  757:  *
  758:  * 1. If any account other than the admin calls the proxy, the call will be forwarded to the implementation, even if
  759:  * that call matches one of the admin functions exposed by the proxy itself.
  760:  * 2. If the admin calls the proxy, it can access the admin functions, but its calls will never be forwarded to the
  761:  * implementation. If the admin tries to call a function on the implementation it will fail with an error that says
  762:  * "admin cannot fallback to proxy target".
  763:  *
  764:  * These properties mean that the admin account can only be used for admin actions like upgrading the proxy or changing
  765:  * the admin, so it's best if it's a dedicated account that is not used for anything else. This will avoid headaches due
  766:  * to sudden errors when trying to call a function from the proxy implementation.
  767:  *
  768:  * Our recommendation is for the dedicated account to be an instance of the {ProxyAdmin} contract. If set up this way,
  769:  * you should think of the `ProxyAdmin` instance as the real administrative interface of your proxy.
  770:  *
  771:  * NOTE: The real interface of this proxy is that defined in `ITransparentUpgradeableProxy`. This contract does not
  772:  * inherit from that interface, and instead the admin functions are implicitly implemented using a custom dispatch
  773:  * mechanism in `_fallback`. Consequently, the compiler will not produce an ABI for this contract. This is necessary to
  774:  * fully implement transparency without decoding reverts caused by selector clashes between the proxy and the
  775:  * implementation.
  776:  *
  777:  * WARNING: It is not recommended to extend this contract to add additional external functions. If you do so, the compiler
  778:  * will not check that there are no selector conflicts, due to the note above. A selector clash between any new function
  779:  * and the functions declared in {ITransparentUpgradeableProxy} will be resolved in favor of the new one. This could
  780:  * render the admin operations inaccessible, which could prevent upgradeability. Transparency may also be compromised.
  781:  */

  783:   /**

  785:    * optionally initialized with `_data` as explained in {ERC1967Proxy-constructor}.
  786:    */

  788:     _changeAdmin(admin_);
