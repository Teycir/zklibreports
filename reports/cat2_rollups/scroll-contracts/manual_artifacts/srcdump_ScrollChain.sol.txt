  262:     function initializeV2() external reinitializer(2) {
  263:         // binary search on lastCommittedBatchIndex
  264:         uint256 index = __lastFinalizedBatchIndex;
  265:         uint256 step = 1;
  266:         unchecked {
  267:             while (committedBatches[index + step] != bytes32(0)) {
  268:                 step <<= 1;
  269:             }
  270:             step >>= 1;
  271:             while (step > 0) {
  272:                 if (committedBatches[index + step] != bytes32(0)) {
  273:                     index += step;
  274:                 }
  275:                 step >>= 1;
  276:             }
  277:         }
  278: 
  279:         miscData = ScrollChainMiscData({
  280:             lastCommittedBatchIndex: uint64(index),
  281:             lastFinalizedBatchIndex: uint64(__lastFinalizedBatchIndex),
  282:             lastFinalizeTimestamp: uint32(block.timestamp),
  283:             flags: 0,
  284:             reserved: 0
  285:         });
  286:     }

  311:     /// @notice Import layer 2 genesis block
  312:     /// @param _batchHeader The header of the genesis batch.
  313:     /// @param _stateRoot The state root of the genesis block.
  314:     function importGenesisBatch(bytes calldata _batchHeader, bytes32 _stateRoot) external {
  315:         // check genesis batch header length
  316:         if (_stateRoot == bytes32(0)) revert ErrorStateRootIsZero();
  317: 
  318:         // check whether the genesis batch is imported
  319:         if (finalizedStateRoots[0] != bytes32(0)) revert ErrorGenesisBatchImported();
  320: 
  321:         (uint256 memPtr, bytes32 _batchHash, , ) = _loadBatchHeader(_batchHeader, 0);
  322: 
  323:         // check all fields except `dataHash` and `lastBlockHash` are zero
  324:         unchecked {
  325:             uint256 sum = BatchHeaderV0Codec.getVersion(memPtr) +
  326:                 BatchHeaderV0Codec.getBatchIndex(memPtr) +
  327:                 BatchHeaderV0Codec.getL1MessagePopped(memPtr) +
  328:                 BatchHeaderV0Codec.getTotalL1MessagePopped(memPtr);
  329:             if (sum != 0) revert ErrorGenesisBatchHasNonZeroField();
  330:         }
  331:         if (BatchHeaderV0Codec.getDataHash(memPtr) == bytes32(0)) revert ErrorGenesisDataHashIsZero();
  332:         if (BatchHeaderV0Codec.getParentBatchHash(memPtr) != bytes32(0)) revert ErrorGenesisParentBatchHashIsNonZero();
  333: 
  334:         committedBatches[0] = _batchHash;
  335:         finalizedStateRoots[0] = _stateRoot;
  336: 
  337:         emit CommitBatch(0, _batchHash);
  338:         emit FinalizeBatch(0, _batchHash, _stateRoot, bytes32(0));
  339:     }

  350:     /// @inheritdoc IScrollChain
  351:     /// @dev This function cannot revert V6 and V7 batches at the same time, so we will assume all batches are V7.
  352:     /// If we need to revert V6 batches, we can downgrade the contract to the previous version and call this function.
  353:     /// @dev During commit batch we only store the last batch hash into storage. As a result, we cannot revert intermediate batches.
  354:     function revertBatch(bytes calldata batchHeader) external onlyOwner {
  355:         uint256 lastBatchIndex = miscData.lastCommittedBatchIndex;
  356:         (uint256 batchPtr, , uint256 startBatchIndex, ) = _loadBatchHeader(batchHeader, lastBatchIndex);
  357:         // only revert v7 batches
  358:         if (BatchHeaderV0Codec.getVersion(batchPtr) < 7) revert ErrorIncorrectBatchVersion();
  359:         // check finalization
  360:         if (startBatchIndex < miscData.lastFinalizedBatchIndex) revert ErrorRevertFinalizedBatch();
  361: 
  362:         // actual revert
  363:         for (uint256 i = lastBatchIndex; i > startBatchIndex; --i) {
  364:             bytes32 hash = committedBatches[i];
  365:             if (hash != bytes32(0)) delete committedBatches[i];
  366:         }
  367:         emit RevertBatch(startBatchIndex + 1, lastBatchIndex);
  368: 
  369:         // update `lastCommittedBatchIndex`
  370:         miscData.lastCommittedBatchIndex = uint64(startBatchIndex);
  371:     }

  373:     /// @inheritdoc IScrollChain
  374:     function finalizeBundlePostEuclidV2(
  375:         bytes calldata batchHeader,
  376:         uint256 totalL1MessagesPoppedOverall,
  377:         bytes32 postStateRoot,
  378:         bytes32 withdrawRoot,
  379:         bytes calldata aggrProof
  380:     ) external override OnlyProver whenNotPaused whenEnforcedBatchNotEnabled {
  381:         uint256 flags = miscData.flags;
  382:         bool isV1MessageFinalized = _decodeBoolFromFlag(flags, V1_MESSAGES_FINALIZED_OFFSET);
  383:         if (!isV1MessageFinalized) {
  384:             if (
  385:                 IL1MessageQueueV1(messageQueueV1).nextUnfinalizedQueueIndex() !=
  386:                 IL1MessageQueueV2(messageQueueV2).firstCrossDomainMessageIndex()
  387:             ) {
  388:                 revert ErrorNotAllV1MessagesAreFinalized();
  389:             }
  390:             miscData.flags = uint8(_insertBoolToFlag(flags, V1_MESSAGES_FINALIZED_OFFSET, true));
  391:         }
  392: 
  393:         _finalizeBundlePostEuclidV2(batchHeader, totalL1MessagesPoppedOverall, postStateRoot, withdrawRoot, aggrProof);
  394:     }

  396:     /// @inheritdoc IScrollChain
  397:     /// @dev We only consider batch version >= 7 here.
  398:     function commitAndFinalizeBatch(
  399:         uint8 version,
  400:         bytes32 parentBatchHash,
  401:         FinalizeStruct calldata finalizeStruct
  402:     ) external OnlyTopLevelCall {
  403:         ScrollChainMiscData memory cachedMiscData = miscData;
  404:         if (!isEnforcedModeEnabled()) {
  405:             (uint256 maxDelayEnterEnforcedMode, uint256 maxDelayMessageQueue) = SystemConfig(systemConfig)
  406:                 .enforcedBatchParameters();
  407:             uint256 firstUnfinalizedMessageTime = IL1MessageQueueV2(messageQueueV2)
  408:                 .getFirstUnfinalizedMessageEnqueueTime();
  409:             if (
  410:                 firstUnfinalizedMessageTime + maxDelayMessageQueue < block.timestamp ||
  411:                 cachedMiscData.lastFinalizeTimestamp + maxDelayEnterEnforcedMode < block.timestamp
  412:             ) {
  413:                 if (cachedMiscData.lastFinalizedBatchIndex < cachedMiscData.lastCommittedBatchIndex) {
  414:                     // be careful with the gas costs, maybe should call revertBatch first.
  415:                     for (
  416:                         uint256 i = cachedMiscData.lastCommittedBatchIndex;
  417:                         i > cachedMiscData.lastFinalizedBatchIndex;
  418:                         --i
  419:                     ) {
  420:                         bytes32 hash = committedBatches[i];
  421:                         if (hash != bytes32(0)) delete committedBatches[i];
  422:                     }
  423:                     emit RevertBatch(
  424:                         cachedMiscData.lastFinalizedBatchIndex + 1,
  425:                         cachedMiscData.lastCommittedBatchIndex
  426:                     );
  427:                 }
  428:                 // explicitly enable enforced batch mode
  429:                 cachedMiscData.flags = uint8(_insertBoolToFlag(cachedMiscData.flags, ENFORCED_MODE_OFFSET, true));
  430:                 // reset `lastCommittedBatchIndex`
  431:                 cachedMiscData.lastCommittedBatchIndex = uint64(cachedMiscData.lastFinalizedBatchIndex);
  432:                 miscData = cachedMiscData;
  433:                 emit UpdateEnforcedBatchMode(true, cachedMiscData.lastCommittedBatchIndex);
  434:             } else {
  435:                 revert ErrorNotInEnforcedBatchMode();
  436:             }
  437:         }
  438: 
  439:         bytes32 batchHash = keccak256(finalizeStruct.batchHeader);
  440:         _commitBatchesFromV7(version, parentBatchHash, batchHash, true);
  441: 
  442:         // finalize with zk proof
  443:         _finalizeBundlePostEuclidV2(
  444:             finalizeStruct.batchHeader,
  445:             finalizeStruct.totalL1MessagesPoppedOverall,
  446:             finalizeStruct.postStateRoot,
  447:             finalizeStruct.withdrawRoot,
  448:             finalizeStruct.zkProof
  449:         );
  450:     }

  515:     /// @dev Caller should make sure bit is smaller than 256.
  516:     function _decodeBoolFromFlag(uint256 flag, uint256 bit) internal pure returns (bool) {
  517:         return (flag >> bit) & 1 == 1;
  518:     }
  519: 
  520:     /// @dev Caller should make sure bit is smaller than 256.
  521:     function _insertBoolToFlag(
  522:         uint256 flag,
  523:         uint256 bit,
  524:         bool value
  525:     ) internal pure returns (uint256) {
  526:         flag = flag ^ (flag & (1 << bit)); // reset value at bit
  527:         if (value) {
  528:             flag |= (1 << bit);
  529:         }
  530:         return flag;
  531:     }

  533:     /// @dev Internal function to do common actions before actual batch finalization.
  534:     function _beforeFinalizeBatch(bytes calldata batchHeader, bytes32 postStateRoot)
  535:         internal
  536:         view
  537:         returns (
  538:             uint256 version,
  539:             bytes32 batchHash,
  540:             uint256 batchIndex,
  541:             uint256 totalL1MessagesPoppedOverall,
  542:             uint256 prevBatchIndex
  543:         )
  544:     {
  545:         if (postStateRoot == bytes32(0)) revert ErrorStateRootIsZero();
  546: 
  547:         ScrollChainMiscData memory cachedMiscData = miscData;
  548:         uint256 batchPtr;
  549:         // compute pending batch hash and verify
  550:         (batchPtr, batchHash, batchIndex, totalL1MessagesPoppedOverall) = _loadBatchHeader(
  551:             batchHeader,
  552:             cachedMiscData.lastCommittedBatchIndex
  553:         );
  554: 
  555:         // make sure don't finalize batch multiple times
  556:         prevBatchIndex = cachedMiscData.lastFinalizedBatchIndex;
  557:         if (batchIndex <= prevBatchIndex) revert ErrorBatchIsAlreadyVerified();
  558: 
  559:         version = BatchHeaderV0Codec.getVersion(batchPtr);
  560:     }

  562:     /// @dev Internal function to do common actions after actual batch finalization.
  563:     function _afterFinalizeBatch(
  564:         uint256 batchIndex,
  565:         bytes32 batchHash,
  566:         uint256 totalL1MessagesPoppedOverall,
  567:         bytes32 postStateRoot,
  568:         bytes32 withdrawRoot,
  569:         bool isV1
  570:     ) internal {
  571:         ScrollChainMiscData memory cachedMiscData = miscData;
  572:         cachedMiscData.lastFinalizedBatchIndex = uint64(batchIndex);
  573:         cachedMiscData.lastFinalizeTimestamp = uint32(block.timestamp);
  574:         miscData = cachedMiscData;
  575:         // @note we do not store intermediate finalized roots
  576:         finalizedStateRoots[batchIndex] = postStateRoot;
  577:         withdrawRoots[batchIndex] = withdrawRoot;
  578: 
  579:         // Pop finalized and non-skipped message from L1MessageQueue.
  580:         _finalizePoppedL1Messages(totalL1MessagesPoppedOverall, isV1);
  581: 
  582:         emit FinalizeBatch(batchIndex, batchHash, postStateRoot, withdrawRoot);
  583:     }

  588:         // Get blob's versioned hash

  594:     /// @dev Internal function to commit one ore more batches after the EuclidV2 upgrade.
  595:     /// @param version The version of the batches (version >= 7).
  596:     /// @param parentBatchHash The hash of parent batch.
  597:     /// @param lastBatchHash The hash of the last committed batch after this call.
  598:     /// @param onlyOne If true, we will only process the first blob.
  599:     function _commitBatchesFromV7(
  600:         uint8 version,
  601:         bytes32 parentBatchHash,
  602:         bytes32 lastBatchHash,
  603:         bool onlyOne
  604:     ) internal {
  605:         if (version < 7) {
  606:             // only accept version >= 7
  607:             revert ErrorIncorrectBatchVersion();
  608:         }
  609: 
  610:         uint256 lastCommittedBatchIndex = miscData.lastCommittedBatchIndex;
  611:         if (parentBatchHash != committedBatches[lastCommittedBatchIndex]) revert ErrorIncorrectBatchHash();
  612:         for (uint256 i = 0; ; i++) {
  613:             bytes32 blobVersionedHash = _getBlobVersionedHash(i);
  614:             if (blobVersionedHash == bytes32(0)) {
  615:                 if (i == 0) revert ErrorBatchIsEmpty();
  616:                 break;
  617:             }
  618: 
  619:             lastCommittedBatchIndex += 1;
  620:             // see comments in `src/libraries/codec/BatchHeaderV7Codec.sol` for encodings
  621:             uint256 batchPtr = BatchHeaderV7Codec.allocate();
  622:             BatchHeaderV0Codec.storeVersion(batchPtr, version);
  623:             BatchHeaderV0Codec.storeBatchIndex(batchPtr, lastCommittedBatchIndex);
  624:             BatchHeaderV7Codec.storeParentBatchHash(batchPtr, parentBatchHash);
  625:             BatchHeaderV7Codec.storeBlobVersionedHash(batchPtr, blobVersionedHash);
  626:             bytes32 batchHash = BatchHeaderV0Codec.computeBatchHash(
  627:                 batchPtr,
  628:                 BatchHeaderV7Codec.BATCH_HEADER_FIXED_LENGTH
  629:             );
  630:             emit CommitBatch(lastCommittedBatchIndex, batchHash);
  631:             parentBatchHash = batchHash;
  632:             if (onlyOne) break;
  633:         }
  634: 
  635:         // Make sure that the batch hash matches the one computed by the batch committer off-chain.
  636:         // This check can fail if:
  637:         // 1. faulty batch producers commit a wrong batch or the local computation is wrong.
  638:         // 2. unexpected `parentBatch` in case commit transactions get reordered
  639:         // 3. two batch producers commit at the same time with the same `parentBatch`.
  640:         if (parentBatchHash != lastBatchHash) {
  641:             revert InconsistentBatchHash(lastCommittedBatchIndex, lastBatchHash, parentBatchHash);
  642:         }
  643:         // only store last batch hash in storage
  644:         committedBatches[lastCommittedBatchIndex] = parentBatchHash;
  645:         miscData.lastCommittedBatchIndex = uint64(lastCommittedBatchIndex);
  646:     }

  648:     /// @dev Internal function to finalize a bundle after the EuclidV2 upgrade.
  649:     /// @param batchHeader The header of the last batch in this bundle.
  650:     /// @param totalL1MessagesPoppedOverall The number of messages processed after this bundle.
  651:     /// @param postStateRoot The state root after this bundle.
  652:     /// @param withdrawRoot The withdraw trie root after this bundle.
  653:     /// @param aggrProof The bundle proof for this bundle.
  654:     function _finalizeBundlePostEuclidV2(
  655:         bytes calldata batchHeader,
  656:         uint256 totalL1MessagesPoppedOverall,
  657:         bytes32 postStateRoot,
  658:         bytes32 withdrawRoot,
  659:         bytes calldata aggrProof
  660:     ) internal {
  661:         // actions before verification
  662:         (uint256 version, bytes32 batchHash, uint256 batchIndex, , uint256 prevBatchIndex) = _beforeFinalizeBatch(
  663:             batchHeader,
  664:             postStateRoot
  665:         );
  666: 
  667:         // L1 message hashes are chained,
  668:         // this hash commits to the whole queue up to and including `totalL1MessagesPoppedOverall-1`
  669:         bytes32 messageQueueHash = totalL1MessagesPoppedOverall == 0
  670:             ? bytes32(0)
  671:             : IL1MessageQueueV2(messageQueueV2).getMessageRollingHash(totalL1MessagesPoppedOverall - 1);
  672: 
  673:         bytes memory publicInputs = abi.encodePacked(
  674:             layer2ChainId,
  675:             messageQueueHash,
  676:             uint32(batchIndex - prevBatchIndex), // numBatches
  677:             finalizedStateRoots[prevBatchIndex], // _prevStateRoot
  678:             committedBatches[prevBatchIndex], // _prevBatchHash
  679:             postStateRoot,
  680:             batchHash,
  681:             withdrawRoot
  682:         );
  683: 
  684:         // verify bundle, choose the correct verifier based on the last batch
  685:         // our off-chain service will make sure all unfinalized batches have the same batch version.
  686:         IRollupVerifier(verifier).verifyBundleProof(version, batchIndex, aggrProof, publicInputs);
  687: 
  688:         // actions after verification
  689:         _afterFinalizeBatch(batchIndex, batchHash, totalL1MessagesPoppedOverall, postStateRoot, withdrawRoot, false);
  690:     }

  692:     /// @dev Internal function to load batch header from calldata to memory.
  693:     /// @param _batchHeader The batch header in calldata.
  694:     /// @param _lastCommittedBatchIndex The index of the last committed batch.
  695:     /// @return batchPtr The start memory offset of loaded batch header.
  696:     /// @return _batchHash The hash of the loaded batch header.
  697:     /// @return _batchIndex The index of this batch.
  698:     /// @return _totalL1MessagesPoppedOverall The number of L1 messages popped after this batch.
  699:     /// @dev This function only works with batches whose hashes are stored in `committedBatches`.
  700:     function _loadBatchHeader(bytes calldata _batchHeader, uint256 _lastCommittedBatchIndex)
  701:         internal
  702:         view
  703:         virtual
  704:         returns (
  705:             uint256 batchPtr,
  706:             bytes32 _batchHash,
  707:             uint256 _batchIndex,
  708:             uint256 _totalL1MessagesPoppedOverall
  709:         )
  710:     {
  711:         // load version from batch header, it is always the first byte.
  712:         uint256 version;
  713:         assembly {
  714:             version := shr(248, calldataload(_batchHeader.offset))
  715:         }
  716: 
  717:         uint256 _length;
  718:         if (version == 0) {
  719:             (batchPtr, _length) = BatchHeaderV0Codec.loadAndValidate(_batchHeader);
  720:         } else if (version <= 2) {
  721:             (batchPtr, _length) = BatchHeaderV1Codec.loadAndValidate(_batchHeader);
  722:         } else if (version <= 6) {
  723:             (batchPtr, _length) = BatchHeaderV3Codec.loadAndValidate(_batchHeader);
  724:         } else {
  725:             (batchPtr, _length) = BatchHeaderV7Codec.loadAndValidate(_batchHeader);
  726:         }
  727: 
  728:         // the code for compute batch hash is the same for V0~V6
  729:         // also the `_batchIndex` and `_totalL1MessagesPoppedOverall`.
  730:         _batchHash = BatchHeaderV0Codec.computeBatchHash(batchPtr, _length);
  731:         _batchIndex = BatchHeaderV0Codec.getBatchIndex(batchPtr);
  732:         // we don't have totalL1MessagesPoppedOverall in V7~
  733:         if (version <= 6) {
  734:             _totalL1MessagesPoppedOverall = BatchHeaderV0Codec.getTotalL1MessagePopped(batchPtr);
  735:         }
  736: 
  737:         if (_batchIndex > _lastCommittedBatchIndex) revert ErrorBatchNotCommitted();
  738: 
  739:         // only check when genesis is imported
  740:         if (committedBatches[_batchIndex] != _batchHash && finalizedStateRoots[0] != bytes32(0)) {
  741:             revert ErrorIncorrectBatchHash();
  742:         }
  743:     }

  745:     /// @param totalL1MessagesPoppedOverall The total number of L1 messages popped in all batches including current batch.
  746:     function _finalizePoppedL1Messages(uint256 totalL1MessagesPoppedOverall, bool isV1) internal {
  747:         if (totalL1MessagesPoppedOverall > 0) {
  748:             if (isV1) {
  749:                 IL1MessageQueueV1(messageQueueV1).finalizePoppedCrossDomainMessage(totalL1MessagesPoppedOverall);
  750:             } else {
  751:                 IL1MessageQueueV2(messageQueueV2).finalizePoppedCrossDomainMessage(totalL1MessagesPoppedOverall);
  752:             }
  753:         }
  754:     }
