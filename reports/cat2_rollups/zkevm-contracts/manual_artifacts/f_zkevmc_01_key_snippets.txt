=== AggLayerGateway verify path (route-selected verifier) ===
  124:     function verifyPessimisticProof(
  125:         bytes calldata publicValues,
  126:         bytes calldata proofBytes
  127:     ) external view {
  128:         /// @dev By protocol the proof should at least have the 4 bytes selector, the other bytes are not part of our protocol
  129:         if (proofBytes.length < 4) {
  130:             revert InvalidProofBytesLength();
  131:         }
  132: 
  133:         bytes4 ppSelector = bytes4(proofBytes[:4]);
  134: 
  135:         AggLayerVerifierRoute memory route = pessimisticVKeyRoutes[ppSelector];
  136:         if (route.verifier == address(0)) {
  137:             revert RouteNotFound(ppSelector);
  138:         } else if (route.frozen) {
  139:             revert RouteIsFrozen(ppSelector);
  140:         }
  141: 
  142:         ISP1Verifier(route.verifier).verifyProof(
  143:             route.pessimisticVKey,
  144:             publicValues,
  145:             proofBytes[4:]
  146:         );
  147:     }
